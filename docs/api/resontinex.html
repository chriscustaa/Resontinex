287    def calculate_weighted_score(self, scenario_scores: Dict[str, float]) -> float:
288        """
289        Advanced weighted scoring with multi-dimensional performance analysis.
290        
291        Incorporates scenario weights, performance history, complexity factors,
292        and adaptive learning for optimal scoring accuracy.
293        """
294        if not scenario_scores:
295            return 0.0
296        
297        total_weighted_score = 0.0
298        total_weight = 0.0
299        performance_bonus = 0.0
300        
301        for scenario_id, score in scenario_scores.items():
302            if scenario_id not in self.scenarios:
303                logging.warning(f"Unknown scenario in scoring: {scenario_id}")
304                continue
305            
306            scenario = self.scenarios[scenario_id]
307            base_weight = scenario.weight
308            
309            # Apply adaptive weight adjustments
310            if self.adaptive_filtering:
311                performance_factor = self._get_performance_factor(scenario_id)
312                complexity_factor = self._get_complexity_factor(scenario)
313                recency_factor = self._get_recency_factor(scenario_id)
314                
315                adjusted_weight = base_weight * performance_factor * complexity_factor * recency_factor
316            else:
317                adjusted_weight = base_weight
318            
319            # Calculate contribution
320            weighted_contribution = score * adjusted_weight
321            total_weighted_score += weighted_contribution
322            total_weight += adjusted_weight
323            
324            # Performance bonus for high-performing scenarios
325            if score > 0.9 and adjusted_weight > self.performance_threshold:
326                performance_bonus += 0.05 * adjusted_weight
327        
328        if total_weight == 0:
329            return 0.0
330        
331        base_score = total_weighted_score / total_weight
332        final_score = min(1.0, base_score + performance_bonus)
333        
334        # Update performance history
335        self._update_performance_history(scenario_scores, final_score)
336        
337        return final_score

403    def _calculate_performance_threshold(self, base_threshold: float, weights: List[float]) -> float:
404        """Calculate adaptive threshold based on weight distribution and performance."""
405        if not weights:
406            return base_threshold
407        
408        percentile_75 = sorted(weights)[int(0.75 * len(weights))]
409        mean_weight = statistics.mean(weights)
410        
411        # Adjust threshold based on distribution characteristics
412        if base_threshold > percentile_75:
413            return min(base_threshold, percentile_75 + 0.1)
414        elif base_threshold < mean_weight * 0.5:
415            return max(base_threshold, mean_weight * 0.6)
416        
417        return base_threshold

419    def _apply_composite_scoring(self, scenarios: Dict[str, Scenario]) -> Dict[str, Scenario]:
420        """Apply composite scoring to reduce scenario set to optimal candidates."""
421        if len(scenarios) <= 5:
422            return scenarios
423        
424        scenario_scores = []
425        
426        for scenario_id, scenario in scenarios.items():
427            composite_score = (
428                scenario.weight * 0.4 +
429                self._get_performance_factor(scenario_id) * 0.3 +
430                (scenario.priority / 10.0) * 0.2 +
431                self._get_recency_factor(scenario_id) * 0.1
432            )
433            scenario_scores.append((scenario_id, scenario, composite_score))
434        
435        # Select top scenarios based on composite score
436        scenario_scores.sort(key=lambda x: x[2], reverse=True)
437        top_scenarios = scenario_scores[:8]  # Keep top 8 scenarios
438        
439        return {item[0]: item[1] for item in top_scenarios}

441    def _get_performance_factor(self, scenario_id: str) -> float:
442        """Get performance factor for scenario based on historical data."""
443        if scenario_id not in self.performance_history:

447        if not history:
448            return 1.0

450        # Calculate moving average of recent performance

452        avg_performance = statistics.mean(recent_scores)
453        
454        # Convert to factor (0.5 - 1.5 range)
455        return 0.5 + avg_performance
